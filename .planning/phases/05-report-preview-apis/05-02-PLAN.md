2. Add `_preview_csv()` helper function after `_validate_preview_path()`:

```python
def _preview_csv(path: Path) -> dict:
    """Read CSV with row cap and pagination metadata.
    
    Returns:
        dict with type, headers, rows, count, total, capped, cap_limit
    """
    import csv as csv_module
    
    total = 0
    rows = []
    headers = []
    
    with open(path, newline='', encoding='utf-8', errors='replace') as f:
        reader = csv_module.DictReader(f)
        headers = reader.fieldnames or []
        for row in reader:
            total += 1
            if len(rows) < MAX_PREVIEW_ROWS:
                rows.append(row)
    
    return {
        "type": "csv",
        "headers": headers,
        "rows": rows,
        "count": len(rows),
        "total": total,
        "capped": total > MAX_PREVIEW_ROWS,
        "cap_limit": MAX_PREVIEW_ROWS
    }
```
