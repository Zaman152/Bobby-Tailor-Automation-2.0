```javascript
// Global state for plans
let allPlans = [];
let selectedProjectId = null;

// Sheet type detection based on sheet name
function getSheetType(sheetName) {
    const upper = (sheetName || '').toUpperCase();
    if (upper.startsWith('A') || upper.includes('FLOOR') || upper.includes('PLAN') || upper.includes('ELEVATION')) return 'architectural';
    if (upper.startsWith('E') || upper.includes('ELECTRICAL') || upper.includes('PANEL')) return 'electrical';
    if (upper.startsWith('M') || upper.includes('MECHANICAL') || upper.includes('HVAC')) return 'mechanical';
    if (upper.includes('SCHEDULE')) return 'schedule';
    return 'other';
}

// Type badge color mapping
function getTypeBadgeStyle(type) {
    const colors = {
        'architectural': '#3b82f6',  // blue
        'electrical': '#f59e0b',      // yellow/amber
        'mechanical': '#f97316',      // orange
        'schedule': '#8b5cf6',        // purple
        'other': '#6b7280'            // gray
    };
    return `background: ${colors[type] || colors.other}; padding: 2px 8px; border-radius: 4px; font-size: 11px; color: white;`;
}

// Load plans for selected project
async function loadProjectPlans() {
    if (!selectedProjectId) {
        alert('Please select a project first');
        return;
    }
    
    const panel = document.getElementById('planSelectionPanel');
    const plansList = document.getElementById('plansList');
    
    panel.style.display = 'block';
    plansList.innerHTML = '<div style="color: #94a3b8; padding: 20px; text-align: center;">Loading plans...</div>';
    
    try {
        const res = await fetch(`/api/projects/${selectedProjectId}/plans`);
        const data = await res.json();
        
        if (data.error) {
            plansList.innerHTML = `<div style="color: #ef4444; padding: 20px;">Error: ${data.error}</div>`;
            return;
        }
        
        allPlans = data.plans || [];
        
        if (allPlans.length === 0) {
            plansList.innerHTML = '<div style="color: #94a3b8; padding: 20px;">No plans found for this project</div>';
            return;
        }
        
        renderPlansList(allPlans);
        updateSelectedCount();
        
    } catch (err) {
        plansList.innerHTML = `<div style="color: #ef4444; padding: 20px;">Failed to load plans: ${err.message}</div>`;
    }
}

// Render plans list with checkboxes
function renderPlansList(plans) {
    const plansList = document.getElementById('plansList');
    const filter = document.getElementById('sheetTypeFilter').value;
    
    const filtered = filter === 'all' ? plans : plans.filter(p => getSheetType(p.sheet_name) === filter);
    
    if (filtered.length === 0) {
        plansList.innerHTML = '<div style="color: #94a3b8; padding: 20px;">No plans match the selected filter</div>';
        return;
    }
    
    plansList.innerHTML = filtered.map(plan => {
        const type = getSheetType(plan.sheet_name);
        return `
            <label style="display: flex; align-items: center; padding: 10px; border-bottom: 1px solid #252a3a; cursor: pointer;" 
                   data-page-id="${plan.page_id}" data-type="${type}">
                <input type="checkbox" class="plan-checkbox" value="${plan.page_id}" 
                       onchange="updateSelectedCount()" style="margin-right: 12px; width: 18px; height: 18px;">
                <span style="flex: 1; color: #f1f5f9;">${plan.sheet_name || 'Unnamed Sheet'}</span>
                <span style="${getTypeBadgeStyle(type)}">${type}</span>
            </label>
        `;
    }).join('');
}

// Toggle all checkboxes
function toggleAllPlans(checked) {
    document.querySelectorAll('.plan-checkbox').forEach(cb => {
        cb.checked = checked;
    });
    updateSelectedCount();
}

// Select none
function selectNone() {
    document.getElementById('selectAllPlans').checked = false;
    toggleAllPlans(false);
}

// Filter plans by type
function filterPlansByType(type) {
    renderPlansList(allPlans);
    updateSelectedCount();
}

// Update selected count display
function updateSelectedCount() {
    const checked = document.querySelectorAll('.plan-checkbox:checked');
    const count = checked.length;
    document.getElementById('selectedCount').textContent = `${count} sheet${count !== 1 ? 's' : ''} selected`;
    document.getElementById('runSelectedBtn').disabled = count === 0;
    
    // Update Select All checkbox state
    const total = document.querySelectorAll('.plan-checkbox').length;
    document.getElementById('selectAllPlans').checked = count === total && total > 0;
}

// Run analysis on selected plans only
async function runSelectedPlans() {
    const checked = document.querySelectorAll('.plan-checkbox:checked');
    const pageIds = Array.from(checked).map(cb => parseInt(cb.value));
    
    if (pageIds.length === 0) {
        alert('Please select at least one plan');
        return;
    }
    
    // Get project name from current selection
    const projectName = document.querySelector('#projectSelect option:checked')?.textContent || 'Project';
    
    try {
        const res = await fetch('/api/run/stackct', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                mode: 'specific',
                project_id: selectedProjectId,
                project_name: projectName,
                page_ids: pageIds
            })
        });
        
        const data = await res.json();
        
        if (data.job_id) {
            // Hide plan selection, show job status
            document.getElementById('planSelectionPanel').style.display = 'none';
            // Start polling job status (use existing function if available)
            if (typeof pollJobStatus === 'function') {
                pollJobStatus(data.job_id);
            } else {
                alert(`Job started: ${data.job_id}`);
            }
        } else {
            alert('Failed to start job: ' + (data.error || 'Unknown error'));
        }
    } catch (err) {
        alert('Failed to start job: ' + err.message);
    }
}
```
