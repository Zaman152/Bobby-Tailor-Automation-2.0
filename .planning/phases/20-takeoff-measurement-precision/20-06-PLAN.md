```python
def run_project(self, pages: list[PageInput], progress_callback=None) -> list[dict]:
    """pages: [{image_path, sheet_name, sheet_type_hint?, page_num}]"""
    all_extracted = []
    for page in pages:
        sheet_type = page.sheet_type_hint or classify_sheet_type_from_text(...)
        extracted = self.run_sheet(page.image_path, page.sheet_name, sheet_type)
        if extracted.get("_skipped"):
            continue
        all_extracted.append(extracted)
    project_type = _detect_project_type(all_extracted)
    all_estimates = [apply_estimation_tables(e, project_type=project_type) for e in all_extracted]
    return all_extracted, all_estimates, project_type
```
