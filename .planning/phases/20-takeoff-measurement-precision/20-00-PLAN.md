```python
class TakeoffPipeline:
    def run_sheet(self, image_path: str, sheet_name: str, sheet_type: str | None = None) -> dict:
        # If sheet_type unknown, classify from PDF text or default measure-only
        passes = plan_passes(sheet_type or "floor_plan")
        if not passes:
            return {"_skipped": True, "sheet_type": "title_sheet", ...}
        count_r, measure_r, schedule_r = {}, {}, None
        for p in passes:
            model = pick_model_for_pass(sheet_type, p, sheet_name)
            result = analyze_drawing(image_path, sheet_name, pass_type=p, model_override=model)
            ...
        return merge_passes(count_r, measure_r, schedule_r)
```
