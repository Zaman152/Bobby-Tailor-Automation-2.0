   ```javascript
   function filterProjects(query) {
     const items = document.querySelectorAll('.project-item');
     const q = query.toLowerCase();
     items.forEach(item => {

       item.style.display = name.includes(q) ? 'flex' : 'none';
     });
   }
   ```

### Task 5: Select All / Type Filter Controls

**File:** `static/app.js`

5.1 Select All checkbox toggles all visible plan checkboxes:
   ```javascript
   document.getElementById('selectAllPlans').addEventListener('change', (e) => {
     const visible = document.querySelectorAll('.plan-item:not([style*="display: none"]) input[type="checkbox"]');
     visible.forEach(cb => cb.checked = e.target.checked);
     updateRunButtonCount();
   });
   ```

5.2 Type filter shows/hides plans by `data-type`:
   ```javascript
   document.getElementById('planTypeFilter').addEventListener('change', (e) => {
     const type = e.target.value;
     document.querySelectorAll('.plan-item').forEach(item => {
       if (!type || item.dataset.type === type) {
         item.style.display = 'flex';
       } else {
         item.style.display = 'none';
       }
     });
     // Uncheck Select All when filter changes
     document.getElementById('selectAllPlans').checked = false;
   });
   ```

5.3 Update "Run Selected Plans (N)" count on any checkbox change:
   ```javascript
   function updateRunButtonCount() {
     const checked = document.querySelectorAll('.plan-item input[type="checkbox"]:checked').length;
     const btn = document.getElementById('runSelectedBtn');
     btn.textContent = `Run Selected Plans (${checked}) →`;
     btn.disabled = checked === 0;
   }
   ```

**Acceptance:** Select All works, filter works, button count updates dynamically.
