### Task 1: Fetch Plans on Preview Click

**File:** `static/app.js`

1.1 Implement `fetchPlans(projectId)` to call Phase 4 API:
   ```javascript
   async function fetchPlans(projectId) {
     const planList = document.getElementById('planList');
     planList.innerHTML = '<p class="loading"><span class="spinner"></span> Loading plans...</p>';
     
     try {
       const res = await fetch(`/api/projects/${projectId}/plans`);
       if (!res.ok) throw new Error(`HTTP ${res.status}`);
       const data = await res.json();
       
       if (data.error) {
         planList.innerHTML = `<p class="error">Error: ${escHtml(data.error)}</p>`;
         return;
       }
       
       renderPlans(data.plans);
     } catch (e) {
       planList.innerHTML = `<p class="error">Failed to load plans: ${escHtml(e.message)}</p>`;
     }
   }
   ```

1.2 Wire to Preview button:
   ```javascript
   document.getElementById('previewPlansBtn').addEventListener('click', async () => {
     if (!selectedProject) return;
     document.getElementById('planSelectionPanel').style.display = 'block';
     await fetchPlans(selectedProject.id);
   });
   ```

**Acceptance:** Clicking Preview fetches plans from API, shows loading state.

### Task 2: Render Plan Checkboxes

**File:** `static/app.js`

2.1 Implement `renderPlans(plans)`:
   ```javascript
   function renderPlans(plans) {
     const planList = document.getElementById('planList');
     
     if (!plans || plans.length === 0) {
       planList.innerHTML = '<p class="empty">No drawing pages found for this project.</p>';
       return;
     }
     
     planList.innerHTML = plans.map((plan, idx) => {
       const badgeClass = getSheetTypeBadgeClass(plan.sheet_type);
       const typeName = formatSheetType(plan.sheet_type);
       return `
         <div class="plan-item" data-type="${plan.sheet_type || 'other'}" data-page-id="${plan.page_id}">
           <input type="checkbox" id="plan-${idx}" checked onchange="updateRunButtonCount()">
           <label for="plan-${idx}" class="plan-label">
             <span class="sheet-name">${escHtml(plan.sheet_name)}</span>
           </label>
           <span class="sheet-type-badge ${badgeClass}">${typeName}</span>
         </div>
       `;
     }).join('');
     
     // Reset Select All to checked (all plans selected by default)
     document.getElementById('selectAllPlans').checked = true;
     updateRunButtonCount();
   }
   ```

2.2 Helper to format sheet type display:
   ```javascript
   function formatSheetType(type) {
     const names = {
       'floor_plan': 'Floor Plan',
       'electrical': 'Electrical',
       'mechanical': 'Mechanical',
       'schedule': 'Schedule',
       'architectural': 'Architectural',
       'structural': 'Structural'
     };
     return names[type] || 'Other';
   }
   ```

**Acceptance:** Plans render as checkboxes with type badges, all checked by default.

### Task 3: Collect Selected Page IDs

**File:** `static/app.js`

3.1 Implement `getSelectedPageIds()`:
   ```javascript
   function getSelectedPageIds() {
     const checked = document.querySelectorAll('.plan-item input[type="checkbox"]:checked');
     return Array.from(checked).map(cb => {
       const item = cb.closest('.plan-item');
       return parseInt(item.dataset.pageId, 10);
     });
   }
   ```

3.2 Check if all plans are selected (for optimization):
   ```javascript
   function allPlansSelected() {
     const total = document.querySelectorAll('.plan-item input[type="checkbox"]').length;
     const checked = document.querySelectorAll('.plan-item input[type="checkbox"]:checked').length;
     return total > 0 && total === checked;
   }
   ```

**Acceptance:** Returns array of selected page IDs.

### Task 4: Wire Run Selected Button

**File:** `static/app.js`

4.1 Implement `runSelectedPlans()`:
   ```javascript
   async function runSelectedPlans() {
     const btn = document.getElementById('runSelectedBtn');
     const pageIds = getSelectedPageIds();
     
     if (pageIds.length === 0) {
       alert('Please select at least one plan to run.');
       return;
     }
     
     btn.disabled = true;
     btn.textContent = '⏳ Starting...';
     
     const body = {
       mode: 'specific',
       project_id: selectedProject.id,
       project_name: selectedProject.name
     };
     
     // Only send page_ids if not all plans selected (optimization)
     if (!allPlansSelected()) {
       body.page_ids = pageIds;
     }
     
     try {
       const res = await fetch('/api/run/stackct', {
         method: 'POST',
         headers: { 'Content-Type': 'application/json' },
         body: JSON.stringify(body)
       });
       const data = await res.json();
       
       if (data.error) {
         alert(`Error: ${data.error}`);
         return;
       }
       
       // Start polling for job status (existing function)
       startPolling(data.job_id, selectedProject.name);
       
       // Optionally hide plan panel and scroll to progress
       document.getElementById('planSelectionPanel').style.display = 'none';
     } catch (e) {
       alert(`Failed to start job: ${e.message}`);
     } finally {
       btn.disabled = false;
       btn.textContent = `Run Selected Plans (${pageIds.length}) →`;
     }
   }
   ```

4.2 Wire button click:
   ```javascript
   document.getElementById('runSelectedBtn').addEventListener('click', runSelectedPlans);
   ```

**Acceptance:** Run button sends `page_ids` to API, job starts, progress polling begins.
