Add `selected_pages` parameter:

```python
def run_pdf_analysis(pdf_path: str, project_name: str = "PDF Project",
                     selected_pages: list[int] | None = None,
                     progress_callback=None) -> dict:
    """
    Process a PDF floor plan directly — no browser needed.
    
    Args:
        pdf_path: Path to PDF file
        project_name: Name for report folder
        selected_pages: 1-indexed page numbers to analyze (None = all pages)
        progress_callback: Optional callback(current, total, sheet)
    """
    ts = datetime.now().strftime("%Y%m%d_%H%M%S")
    safe = project_name.replace(" ", "_").replace("/", "-")
    img_dir = Path(SCREENSHOTS_DIR) / f"{safe}_{ts}"
    img_dir.mkdir(parents=True, exist_ok=True)

    doc = fitz.open(pdf_path)
    total_pages = len(doc)
    doc.close()
    
    # Determine which pages to process
    if selected_pages:
        # Convert 1-indexed to 0-indexed, filter valid range
        pages_to_process = [p - 1 for p in selected_pages if 1 <= p <= total_pages]
    else:
        pages_to_process = list(range(total_pages))
    
    total = len(pages_to_process)
    logger.info(f"Processing PDF ({total} of {total_pages} pages): {pdf_path}")

    all_extracted = []
    all_estimates = []

    for idx, i in enumerate(pages_to_process):
        sheet = _sheet_name(pdf_path, i)
        logger.info(f"[{idx+1}/{total}] {sheet}")
        if progress_callback:
            progress_callback(idx + 1, total, sheet)

        img_path = _page_to_image(pdf_path, i, str(img_dir))
        extracted = analyze_drawing(img_path, sheet)
        extracted["_page_num"] = i + 1  # Store 1-indexed page number
        all_extracted.append(extracted)

        if "error" not in extracted:
            all_estimates.extend(apply_estimation_tables(extracted))

    return generate_report(project_name, all_extracted, all_estimates)
```

@app.route("/api/pdf/run", methods=["POST"])
def run_pdf_from_upload():
    """Start PDF analysis job from a previous upload."""
    data = request.json or {}
    upload_id = data.get("upload_id")
    project_name = data.get("project_name")
    selected_pages = data.get("selected_pages")  # list[int] or None
    
    if not upload_id:

        return jsonify({"error": "Upload not found or expired"}), 404
    
    pdf_path = upload["save_path"]

### 3. Update PDF Job Function

**File:** `app.py`

Add new job function that passes `selected_pages`:

```python
def _pdf_job_v2(job_id: str, pdf_path: str, project_name: str, selected_pages: list[int] | None):
    """Background thread for PDF analysis with page selection."""
    from pdf_analyzer import run_pdf_analysis

    def progress(current, total, sheet):
        pct = int(current / total * 100) if total else 0
        jobs[job_id]["progress"] = pct
        jobs[job_id]["log"].append(f"[{current}/{total}] Analyzing {sheet}...")

    jobs[job_id]["status"] = "running"
    pages_desc = f"pages {selected_pages}" if selected_pages else "all pages"
    jobs[job_id]["log"].append(f"Starting PDF analysis: {project_name} ({pages_desc})")
    
    try:
        result = run_pdf_analysis(
            pdf_path, 
            project_name, 
            selected_pages=selected_pages,
            progress_callback=progress
        )
        jobs[job_id]["status"] = "done"
        jobs[job_id]["result"] = result
        total = result.get("total_line_items", 0)
        jobs[job_id]["progress"] = 100
        sheets = result.get("sheets_processed", 0)
        jobs[job_id]["log"].append(f"Done — {sheets} sheets, {total} takeoff items extracted")
    except Exception as e:
        logger.exception("PDF job failed")
        jobs[job_id]["status"] = "error"
        jobs[job_id]["error"] = "The job failed. Check server logs for details."
```
