Minimal job dict builder helper:
```python
def make_job(status="done", **kwargs):
    base = {
        "status": status,
        "project": kwargs.pop("project", "Test Project"),
        "mode": "all",
        "mode_detail": "full",
        "started_at": "2026-06-01T10:00:00",
        "finished_at": "2026-06-01T10:05:00",
        "progress": 100,
        "log": [{"timestamp": "2026-06-01T10:00:01", "type": "info", "message": "done"}],
        "result": kwargs.pop("result", {}),
        "error": kwargs.pop("error", None),
        "warning": kwargs.pop("warning", None),
        "linked_sheets_count": 0,
        "sheets_completed": [],
    }
    base.update(kwargs)
    return base
```
