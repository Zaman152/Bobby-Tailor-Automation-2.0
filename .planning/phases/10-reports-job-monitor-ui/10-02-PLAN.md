# Plan 10-02: Job Monitor Page/Panel

## Overview

**Goal**: Build a dedicated job monitor view showing progress bar, currently analyzing sheet, and scrollable per-sheet log with timestamped entries.

**Master Reference**: §8.4 (Page: Active Job Live Monitor)

**Wave Dependencies**:
- Phase 5 (API endpoints): `/api/status/<job_id>` for polling job state
- Phase 7 (Shell & Navigation): Page routing, sidebar mini-card slot
- Phase 8 (Page components): Button styling, status badges, progress indicators

## Success Criteria

1. Dedicated job monitor view displays when a job is running
2. Progress bar shows percentage and sheet count (8 / 12)
3. Currently analyzing indicator shows current sheet name
4. Sheet log shows per-sheet status with metrics (✓ done, ⟳ analyzing, ○ pending)
5. Log console shows timestamped entries with auto-scroll
6. Cancel job button stops the running job
7. Auto-navigates to Reports page when job completes

### Task 3: Sheet Log List Component
**File**: `static/app.js`

```javascript
function renderSheetLog(sheets) {
  const container = document.getElementById('monitorSheetLog');
  
  container.innerHTML = sheets.map(sheet => `
    <div class="sheet-row ${sheet.status}">
      <span class="sheet-status">${getStatusIcon(sheet.status)}</span>
      <span class="sheet-code">${sheet.code}</span>
      <span class="sheet-name">${sheet.name}</span>
      <span class="sheet-metrics">
        ${sheet.status === 'done' 
          ? `${sheet.measurements} meas  ${sheet.rooms} rooms  ${sheet.components} comp`
          : sheet.status === 'analyzing' 
            ? 'analyzing...'
            : 'pending'
        }
      </span>
    </div>
  `).join('');
}

function getStatusIcon(status) {
  switch (status) {
    case 'done': return '✓';
    case 'analyzing': return '⟳';
    case 'pending': return '○';
    default: return '○';
  }
}
```

**Acceptance**:
- Sheet rows show correct status icon
- Done sheets show metrics
- Analyzing sheet shows "analyzing..."
- Pending sheets show "pending"

function appendLogEntry(timestamp, message) {
  const console = document.getElementById('monitorLogConsole');
  const entry = document.createElement('div');
  entry.className = 'log-entry';
  entry.innerHTML = `<span class="log-time">[${timestamp}]</span> ${escapeHtml(message)}`;

### Task 5: Job Status Polling
**File**: `static/app.js`

```javascript
let jobPollInterval = null;

function startJobPolling(jobId) {
  // Clear any existing interval
  if (jobPollInterval) clearInterval(jobPollInterval);
  
  // Immediate first poll
  pollJobStatus(jobId);
  
  // Poll every 1 second
  jobPollInterval = setInterval(() => pollJobStatus(jobId), 1000);
}

async function pollJobStatus(jobId) {
  try {
    const res = await fetch(`/api/status/${jobId}`);
    const data = await res.json();
    
    updateMonitorUI(data);
    
    // Stop polling if job is done or error
    if (data.status === 'done' || data.status === 'error') {
      stopJobPolling();
      handleJobCompletion(data);
    }
  } catch (err) {
    console.error('Poll failed:', err);
  }
}

function stopJobPolling() {
  if (jobPollInterval) {
    clearInterval(jobPollInterval);
    jobPollInterval = null;
  }
}

function updateMonitorUI(data) {
  // Update header
  document.getElementById('monitorProjectName').textContent = data.project || '—';
  document.getElementById('monitorJobId').textContent = `Job: ${data.id}`;
  document.getElementById('monitorStatus').textContent = `● ${data.status.toUpperCase()}`;
  document.getElementById('monitorStatus').className = `status-badge ${data.status}`;
  
  // Update progress
  updateProgress(data.progress || 0, data.completed_sheets || 0, data.total_sheets || 0);
  
  // Update current sheet
  if (data.current_sheet) {
    document.getElementById('monitorCurrentSheet').innerHTML = 
      `Currently analyzing: <strong>${escapeHtml(data.current_sheet)}</strong>`;
  }
  
  // Update sheet log if available
  if (data.sheet_log) {
    renderSheetLog(data.sheet_log);
  }
  
  // Update log console
  if (data.log && data.log.length > 0) {
    renderLogConsole(data.log);
  }
}
```

**Acceptance**:
- Polling starts when job begins
- UI updates every 1 second
- Polling stops when job ends
- All UI sections update from poll data

### Task 6: Job Completion Handling
**File**: `static/app.js`

```javascript
function handleJobCompletion(data) {
  if (data.status === 'done') {
    // Update status badge
    document.getElementById('monitorStatus').textContent = '✓ COMPLETED';
    document.getElementById('monitorStatus').className = 'status-badge done';
    
    // Show completion message
    appendLogEntry(getCurrentTime(), 'Job completed successfully');
    
    // Navigate to reports after short delay
    setTimeout(() => {
      navigate('reports');
      refreshReports();
    }, 2000);
    
  } else if (data.status === 'error') {
    // Update status badge
    document.getElementById('monitorStatus').textContent = '✗ ERROR';
    document.getElementById('monitorStatus').className = 'status-badge error';
    
    // Show error message
    appendLogEntry(getCurrentTime(), `Error: ${data.error || 'Unknown error'}`);
  }
  
  // Hide cancel button
  document.getElementById('cancelJobBtn').style.display = 'none';
}
```

**Acceptance**:
- Status badge updates on completion
- Completion message appears in log
- Auto-navigates to Reports after 2 seconds on success
- Error status displays with error message

### Task 7: Cancel Job Functionality
**File**: `static/app.js`

```javascript
async function cancelJob() {
  const jobId = currentJobId;
  if (!jobId) return;
  
  if (!confirm('Are you sure you want to cancel this job?')) return;
  
  try {
    const res = await fetch(`/api/cancel/${jobId}`, { method: 'POST' });
    const data = await res.json();
    
    if (data.success) {
      appendLogEntry(getCurrentTime(), 'Job cancelled by user');
      stopJobPolling();
      document.getElementById('monitorStatus').textContent = '⊘ CANCELLED';
      document.getElementById('monitorStatus').className = 'status-badge cancelled';
      document.getElementById('cancelJobBtn').style.display = 'none';
    }
  } catch (err) {
    console.error('Cancel failed:', err);
  }
}
```

**Note**: This requires `/api/cancel/<job_id>` endpoint from Phase 5.

**Acceptance**:
- Confirmation dialog before cancel
- Cancel request sent to API
- UI updates to cancelled state
- Polling stops

### Task 9: Navigation Integration
**File**: `static/app.js`

```javascript
function showJobMonitor(jobId) {
  currentJobId = jobId;
  
  // Navigate to monitor page
  navigate('monitor');
  
  // Reset UI
  document.getElementById('monitorSheetLog').innerHTML = '';
  document.getElementById('monitorLogConsole').innerHTML = '';
  document.getElementById('cancelJobBtn').style.display = 'block';
  
  // Start polling
  startJobPolling(jobId);
}

// Update navigate() to handle 'monitor' page
function navigate(page) {
  // Hide all pages
  document.querySelectorAll('.page').forEach(p => p.style.display = 'none');
  
  // Show target page
  const pageEl = document.getElementById(`${page}Page`);
  if (pageEl) pageEl.style.display = 'block';
  
  // Update nav active state
  document.querySelectorAll('.nav-links li').forEach(li => li.classList.remove('active'));
  const navItem = document.querySelector(`.nav-links li[data-page="${page}"]`);
  if (navItem) navItem.classList.add('active');
}
```

**Acceptance**:
- `showJobMonitor()` called when job starts
- Monitor page displays
- UI resets for new job
- Polling starts automatically
